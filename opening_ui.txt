// ══════════════════════════════════════════════════════
//  BGM 控制 (Điều khiển BGM)
// ══════════════════════════════════════════════════════
(function initBGM() {
  const audio = document.getElementById('bgm-audio');
  const btn   = document.getElementById('bgm-toggle');
  let playing = true;

  function tryPlay() {
    audio.play().catch(() => {});
  }

  // 尝试自动播放 (Thử tự động phát)
  tryPlay();

  // 浏览器可能阻止自动播放，首次用户交互时恢复 (Trình duyệt có thể chặn tự động phát, khôi phục khi người dùng tương tác lần đầu)
  function resumeOnInteraction() {
    if (playing && audio.paused) tryPlay();
    document.removeEventListener('click', resumeOnInteraction);
    document.removeEventListener('keydown', resumeOnInteraction);
  }
  document.addEventListener('click', resumeOnInteraction);
  document.addEventListener('keydown', resumeOnInteraction);

  btn.addEventListener('click', (e) => {
    e.stopPropagation();
    playing = !playing;
    if (playing) {
      tryPlay();
      btn.textContent = 'Bật Nhạc';
      btn.classList.remove('muted');
    } else {
      audio.pause();
      btn.textContent = 'Tắt Nhạc';
      btn.classList.add('muted');
    }
  });
})();

// ══════════════════════════════════════════════════════
//  粒子画布（飘散的金色尘埃） (Canvas hạt bụi vàng bay lơ lửng)

(function initParticles() {
  const canvas = document.getElementById('particle-canvas');
  const ctx    = canvas.getContext('2d');
  let W, H, particles;

  function resize() {
    W = canvas.width  = window.innerWidth;
    H = canvas.height = window.innerHeight;
  }

  function createParticles() {
    particles = Array.from({ length: 55 }, () => ({
      x:    Math.random() * W,
      y:    Math.random() * H,
      r:    0.4 + Math.random() * 1.2,
      vx:   (Math.random() - 0.5) * 0.18,
      vy:   -0.08 - Math.random() * 0.18,
      a:    Math.random(),
      va:   (Math.random() - 0.5) * 0.004,
    }));
  }

  function draw() {
    ctx.clearRect(0, 0, W, H);
    for (const p of particles) {
      p.x  += p.vx;
      p.y  += p.vy;
      p.a  += p.va;
      if (p.a < 0) { p.a = 0; p.va = Math.abs(p.va); }
      if (p.a > 1) { p.a = 1; p.va = -Math.abs(p.va); }
      if (p.y < -10) { p.y = H + 10; p.x = Math.random() * W; }
      if (p.x < -10) { p.x = W + 10; }
      if (p.x > W + 10) { p.x = -10; }

      ctx.beginPath();
      ctx.arc(p.x, p.y, p.r, 0, Math.PI * 2);
      ctx.fillStyle = `hsla(42, 65%, 60%, ${p.a * 0.55})`;
      ctx.fill();
    }
    requestAnimationFrame(draw);
  }

  resize();
  createParticles();
  draw();
  window.addEventListener('resize', () => { resize(); createParticles(); });
})();

function renderStats() {
  return STAT_KEYS.map(key => {
    const val = state.stats[key];
    const pct = ((val - BASE_VAL) / TOTAL_PTS) * 100;
    return `
      <div class="stat-row" id="row-${key}">
        <div class="stat-name">${key}</div>
        <div class="stat-bar-wrap">
          <div class="stat-bar" id="bar-${key}" style="width:${Math.max(pct,0)}%"></div>
        </div>
        <div class="stat-val" id="val-${key}">${val}</div>
        <div class="stat-btns">
          <button class="stat-btn" id="minus-${key}" onclick="adjustStat('${key}',-1)" aria-label="Giảm ${key}">−</button>
          <button class="stat-btn" id="plus-${key}"  onclick="adjustStat('${key}',+1)"  aria-label="Tăng ${key}">＋</button>
        </div>
      </div>`;
  }).join('');
}

function updateProgress() {
  PAGES.forEach((_,i) => {
    const d = document.getElementById('dot-' + i);
    d.className = 'prog-dot' + (i === currentPage ? ' active' : i < currentPage ? ' done' : '');
  });
  document.getElementById('page-counter').textContent = `${currentPage+1} / ${PAGES.length}`;
  document.getElementById('btn-prev').disabled = currentPage === 0;

  const curId = PAGES[currentPage].id;
  const hideNext = curId === 'choose_path' || curId === 'transmigrate';
  const isLast = currentPage === PAGES.length - 1;
  document.getElementById('btn-next').style.display    = (isLast || hideNext) ? 'none' : '';
  document.getElementById('btn-confirm').style.display = isLast ? '' : 'none';
}

//  导航 (Điều hướng)
// ══════════════════════════════════════════════════════
function navigate(dir) {
  if (PAGES[currentPage].id === 'player1') syncPlayer1();

  const oldPage = document.getElementById('page-' + PAGES[currentPage].id);
  oldPage.classList.remove('active');
  oldPage.classList.add('exit');

  const curId = PAGES[currentPage].id;

  if (dir === 1 && curId === 'choose_path') {
    if (state.path === 'transmigrate') {
      currentPage = PAGES.findIndex(p => p.id === 'transmigrate');
    } else {
      currentPage = PAGES.findIndex(p => p.id === 'player1');
    }
  } else if (dir === -1 && (curId === 'transmigrate' || curId === 'player1')) {
    currentPage = PAGES.findIndex(p => p.id === 'choose_path');
  } else {
    currentPage = Math.max(0, Math.min(PAGES.length - 1, currentPage + dir));
  }

  setTimeout(() => { oldPage.classList.remove('exit'); }, 600);

  const newPage = document.getElementById('page-' + PAGES[currentPage].id);
  // 强制重排以触发动画 (Buộc reflow để kích hoạt hoạt ảnh)
  void newPage.offsetWidth;
  newPage.classList.add('active');

  updateProgress();

  if (PAGES[currentPage].id === 'player2') {
    document.getElementById('stats-panel').innerHTML = renderStats();
    document.getElementById('pts-left').textContent  = state.pointsLeft;
    updateStatButtons();
  }
}

function adjustStat(key, delta) {
  if (delta > 0 && state.pointsLeft <= 0) return;
  if (delta < 0 && state.stats[key] <= BASE_VAL) return;
  state.stats[key]  += delta;
  state.pointsLeft  -= delta;

  document.getElementById('val-' + key).textContent = state.stats[key];

  const ptsEl = document.getElementById('pts-left');
  ptsEl.textContent = state.pointsLeft;
  ptsEl.classList.remove('flash');
  void ptsEl.offsetWidth;
  ptsEl.classList.add('flash');

  const pct = ((state.stats[key] - BASE_VAL) / TOTAL_PTS) * 100;
  document.getElementById('bar-' + key).style.width = Math.max(pct, 0) + '%';
  updateStatButtons();
}

function updateStatButtons() {
  STAT_KEYS.forEach(key => {
    const plus  = document.getElementById('plus-'  + key);
    const minus = document.getElementById('minus-' + key);
    if (plus)  plus.disabled  = state.pointsLeft <= 0;
    if (minus) minus.disabled = state.stats[key] <= BASE_VAL;
  });
}

function validate() {
  syncPlayer1();
  if (!state.gender)      { alert('Vui lòng chọn giới tính'); return false; }
  if (!state.appearance)  { alert('Vui lòng điền miêu tả ngoại hình'); return false; }
  if (!state.personality) { alert('Vui lòng điền miêu tả tính cách'); return false; }
  if (!state.openingId)   { alert('Vui lòng chọn khởi cục'); return false; }
  const oc = OPENING_CHOICES.find(c => c.id === state.openingId);
  if (oc && oc.editable && !state.openingCustom) { alert('Vui lòng điền miêu tả khởi cục tự định nghĩa'); return false; }
  if (!state.kungfu)      { alert('Vui lòng chọn võ học ban đầu'); return false; }
  return true;
}

        'Côn Thương':     { 'Sức Mạnh': 0.3, 'Tốc Độ': 0.3 },
        'Kiếm':       { 'Tốc Độ': 0.4, 'Thân Pháp': 0.3 },
        'Đao':       { 'Sức Mạnh': 0.4, 'Phòng Ngự': 0.3 },
        'Ám Khí':     { 'Tốc Độ': 0.4, 'Thân Pháp': 0.4 },
        'Nội Công Tích Lũy': { 'Tu Vi Nội Công': 0.8, 'Phòng Ngự': 0.2 },
      };
      const attrBonus = { 'Sức Mạnh': 0, 'Tốc Độ': 0, 'Phòng Ngự': 0, 'Tu Vi Nội Công': 0, 'Thân Pháp': 0 };
      for (const sk of STAT_KEYS) {
        const sv = (state.stats[sk] || 0) + (kfBskBonus[sk] || 0);
        const mp = BMAP[sk];
        if (!mp) continue;
        for (const at in mp) {
          attrBonus[at] += Math.floor(sv * mp[at]);
        }
      }

      const SứcMạnh     = 10 + attrBonus['Sức Mạnh']     + kfAttrBonus['Sức Mạnh']     + equipBonus['Sức Mạnh'];
      const TốcĐộ     = 10 + attrBonus['Tốc Độ']     + kfAttrBonus['Tốc Độ']     + equipBonus['Tốc Độ'];
      const PhòngNgự   = 10 + attrBonus['Phòng Ngự']   + kfAttrBonus['Phòng Ngự']   + equipBonus['Phòng Ngự'];
      const TuViNộiCông = 10 + attrBonus['Tu Vi Nội Công'] + kfAttrBonus['Tu Vi Nội Công'] + equipBonus['Tu Vi Nội Công'];
      const ThânPháp     = 10 + attrBonus['Thân Pháp']     + kfAttrBonus['Thân Pháp']     + equipBonus['Thân Pháp'];

      _.set(data, 'stat_data.Người Chơi.Thuộc Tính.Sức Mạnh',       SứcMạnh);
      _.set(data, 'stat_data.Người Chơi.Thuộc Tính.Tốc Độ',         TốcĐộ);
      _.set(data, 'stat_data.Người Chơi.Thuộc Tính.Phòng Ngự',      PhòngNgự);
      _.set(data, 'stat_data.Người Chơi.Thuộc Tính.Tu Vi Nội Công', TuViNộiCông);
      _.set(data, 'stat_data.Người Chơi.Thuộc Tính.Thân Pháp',       ThânPháp);

      const HuyếtLượngGiớiHạn = 300 + CấpĐộ * 60 + SứcMạnh * 20 + VõCôngHuyếtLượngGiaThành + TrangBịHuyếtLượngGiaThành;
      const NộiLựcGiớiHạn = 300 + CấpĐộ * 60 + TuViNộiCông * 20 + VõCôngNộiLựcGiaThành + TrangBịNộiLựcGiaThành;

      _.set(data, 'stat_data.Người Chơi.Trạng Thái Cốt Lõi.Lượng Máu.Giới Hạn', HuyếtLượngGiớiHạn);
      _.set(data, 'stat_data.Người Chơi.Trạng Thái Cốt Lõi.Lượng Máu.Hiện Tại', HuyếtLượngGiớiHạn);
      _.set(data, 'stat_data.Người Chơi.Trạng Thái Cốt Lõi.Nội Lực.Giới Hạn', NộiLựcGiớiHạn);
      _.set(data, 'stat_data.Người Chơi.Trạng Thái Cốt Lõi.Nội Lực.Hiện Tại', NộiLựcGiớiHạn);

      function expToNext(lv) {
        if (lv <= 30)  return lv * 60;
        if (lv <= 50)  return lv * 90;
        if (lv <= 70)  return lv * 150;
        if (lv <= 99)  return lv * 210;
        if (lv <= 120) return lv * 450;
        return lv * 600;
      }

      _.set(data, 'stat_data.Người Chơi.Thông Tin Cơ Bản.Cấp Độ', CấpĐộ);
      _.set(data, 'stat_data.Người Chơi.Thông Tin Cơ Bản.Kinh Nghiệm', 0);
      _.set(data, 'stat_data.Người Chơi.Thông Tin Cơ Bản.Kinh Nghiệm Cấp Tiếp Theo', expToNext(CấpĐộ));

      console.log(`[Giao Diện Khởi Đầu] Trạng thái cốt lõi ban đầu: Giới Hạn Máu=${HuyếtLượngGiớiHạn}, Giới Hạn Nội Lực=${NộiLựcGiớiHạn}`);
    }

    let allKungfuRecord;
    if (_kungfuLibCache && typeof _kungfuLibCache === 'object' && Object.keys(_kungfuLibCache).length > 5) {
      allKungfuRecord = Object.fromEntries(
        Object.values(_kungfuLibCache).map(k => [k['Tên Võ Công'] || k['名字'], k['Phân Loại'] || k['类别'] || 'Phổ Thông'])
      );
      console.log('[Giao Diện Khởi Đầu] Nguồn thư viện Võ công: Bộ đệm bộ nhớ, tổng cộng', Object.keys(allKungfuRecord).length, 'cái');
    } else {
      const getAllKungfu = _findGetAllKungfu();
      if (typeof getAllKungfu === 'function') {
        const lib = getAllKungfu();
        _kungfuLibCache = lib;
        allKungfuRecord = Object.fromEntries(
          Object.values(lib).map(k => [k['Tên Võ Công'] || k['名字'], k['Phân Loại'] || k['类别'] || 'Phổ Thông'])
        );
        console.log('[Giao Diện Khởi Đầu] Nguồn thư viện Võ công: WX_GET_ALL_KUNGFU, tổng cộng', Object.keys(allKungfuRecord).length, 'cái');
      } else {
        try {
          const priorData = await Mvu.getMvuData({ type: 'message', message_id: 0 });
          const priorKungfu = priorData?.stat_data?.['Võ Công'];
          if (priorKungfu && Object.keys(priorKungfu).length > 5) {
            allKungfuRecord = priorKungfu;
            console.log('[Giao Diện Khởi Đầu] Nguồn thư viện Võ công: MVU đã bơm trước, tổng cộng', Object.keys(allKungfuRecord).length, 'cái');
          } else {
            throw new Error('Thư viện Võ công trong MVU không đầy đủ');
          }
        } catch(mvuErr) {
          console.warn('[Giao Diện Khởi Đầu] Dự phòng thư viện Võ công (Chỉ 5 võ công đầu):', mvuErr.message);
          allKungfuRecord = Object.fromEntries(KUNGFU_LIST.map(k => [k.id, 'Phổ Thông']));
        }
      }
    }
    _.set(data, 'stat_data.Võ Công', allKungfuRecord);

    await Mvu.replaceMvuData(data, { type: 'message', message_id: 0 });

    // ── 清空世界书"勿动"条目内容（避免上一次魂穿残留影响新角色）(Xoá trắng mục Thế giới thư "勿动" - tránh tàn dư Hồn xuyên lần trước ảnh hưởng nhân vật mới) ──
    try {
      const _p = window.parent;
      const _getNames = _p.getCharWorldbookNames || window.getCharWorldbookNames;
      const _getWB    = _p.getWorldbook           || window.getWorldbook;
      const _repWB    = _p.replaceWorldbook        || window.replaceWorldbook;
      if (_getNames && _getWB && _repWB) {
        const _books = _getNames('current');
        if (_books?.primary) {
          const _entries = await _getWB(_books.primary);
          const _target = _entries.find(e => e.name === '勿动');
          if (_target) {
            _target.content = '';
            await _repWB(_books.primary, _entries, { render: 'debounced' });
          }
        }
      }
    } catch(e) { console.warn('[Giao Diện Khởi Đầu] Bỏ qua xoá "勿动":', e.message); }

    fillInput();

    showSuccess();

    await writeWorldbook();

  } catch(e) {
    console.error('[Giao Diện Khởi Đầu] Lỗi:', e);
    btn.disabled = false;
    btn.innerHTML = `
      <svg style="width:16px;height:16px;stroke:currentColor;fill:none;stroke-width:1.5;flex-shrink:0;" viewBox="0 0 24 24">
        <path d="M5 12h14M12 5l7 7-7 7"/>
      </svg>
      Đạp Nhập Giang Hồ`;
    alert('Đã xảy ra lỗi: ' + e.message);
  }
}

async function writeWorldbook() {
  try {
    const parent = window.parent;
    if (!parent) return;

    const getNames     = parent.getCharWorldbookNames  || window.getCharWorldbookNames;
    const createEntries = parent.createWorldbookEntries || window.createWorldbookEntries;
    if (!getNames || !createEntries) {
      console.warn('[Giao Diện Khởi Đầu] API createWorldbookEntries không khả dụng, bỏ qua');
      return;
    }

    const books = getNames('current');
    if (!books?.primary) {
      console.warn('[Giao Diện Khởi Đầu] Thẻ nhân vật chưa liên kết Thế giới thư chính, bỏ qua');
      return;
    }

    const entryName = `${state.gender}·${state.kungfu}·Tự Định Nghĩa`;
    const content   = `Về {{user}}\nNgoại hình: ${state.appearance}\nTính cách: ${state.personality}`;

    await createEntries(books.primary, [{
      name:     entryName,
      strategy: { type: 'constant', keys: [] },
      position: { type: 'before_character_definition', order: 100 },
      content:  content,
      enabled:  true,
      probability: 100,
      recursion: { prevent_incoming: true, prevent_outgoing: true, delay_until: null },
      effect:   { sticky: null, cooldown: null, delay: null },
    }]);

    showWorldbookNotice(entryName);

  } catch(e) {
    console.warn('[Giao Diện Khởi Đầu] Bỏ qua ghi Thế giới thư:', e.message);
  }
}

function showWorldbookNotice(entryName) {
  const overlay = document.getElementById('wb-notice-overlay');
  const text    = document.getElementById('wb-notice-text');
  if (text) text.textContent =
    `Đã tạo mục Thế giới thư 「${entryName}」 (Đèn xanh · Luôn kích hoạt). ` +
    `Mục này sẽ không tự động xóa——nếu cần chơi lại từ đầu, vui lòng xóa thủ công mục này trong Thế giới thư trước khi tạo cuộc trò chuyện mới.`;
  if (overlay) {
    setTimeout(() => { overlay.style.display = 'flex'; }, 1200);
  }
}

document.getElementById('wb-notice-overlay').addEventListener('click', function(e) {
  if (e.target === this) this.style.display = 'none';
});

function fillInput() {
  const docs = [];
  try { if (window.top?.document)    docs.push(window.top.document);    } catch(e){}
  try { if (window.parent?.document) docs.push(window.parent.document); } catch(e){}
  docs.push(document);

  let textarea = null;
  for (const doc of docs) {
    try {
      textarea = doc.getElementById('send_textarea');
      if (textarea) break;
    } catch(e) {}
  }

  const oc = OPENING_CHOICES.find(c => c.id === state.openingId);
  const msg = oc && oc.editable
    ? `Dựa theo thiết lập khởi cục: ${state.openingCustom}, bắt đầu đi.`
    : `Dựa theo thiết lập của ${oc.name}, bắt đầu đi.`;

  if (textarea) {
    textarea.value = msg;
    textarea.dispatchEvent(new Event('input',  { bubbles: true }));
    textarea.dispatchEvent(new Event('change', { bubbles: true }));
    textarea.dispatchEvent(new KeyboardEvent('keyup', { bubbles: true }));
    textarea.focus();
  }
}

function showSuccess() {
  const overlay = document.getElementById('success-overlay');
  overlay.classList.add('show');
}
